					变量名字
						*>由字母(区分大小写) 数字 或下划线组成
							int A;不等于 int a;
							例子: 
								int a;
								int A;
								int #;//错误，会报错 
								
						*>不能以数字为开头，最好不要以下划线开头
							int a2;
							int 2a;//错误 
							int _a;//虽然有些编译器可以通过，但最好不要这样写 
							
						*>变量名字最好有意义 
							int a; 
							int age;--->年龄 
							
						*>变量名字不能用关键字
							int int;//虽然符合全称，但是编译器默认后面int为变量名，从而报错
							
						*>变量名字最好不要太长:不要超过63个字母
