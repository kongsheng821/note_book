printf的特殊显示：
	显示整数：
			%d  %hd %hhd %ld %lld
		特殊：
			%md . m是个数字， 代表当前数据的宽度
			m : 正数 ，右对齐
			m : 负数 ，左对齐
			m如果小于实际的长度，则无效
			例子：
					int a = 123456;
					printf("%10d\\n",a);
		显示小数：
		%m.nf
				m是个数字，代表当前数据的宽度（包含小数点）
				n是个数字，代表当前数据的小数位
			float 发=123.456
			%10.2f;   //  123.46
